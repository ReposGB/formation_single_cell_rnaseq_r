# Répertoire racine contenant tous les échantillons
input_dir <- "GSE155698_RAW"

# Liste des sous-répertoires (échantillons)
sample_dirs <- list.dirs(input_dir, recursive = FALSE)

# Créer une liste d’objets Seurat
seurat_list <- lapply(sample_dirs, function(dir) {
  
  sample_name <- basename(dir)  # nom de l’échantillon = nom du dossier

  print(paste("read sample ", sample_name))
  
  cur_data <- Read10X(data.dir = file.path(dir, 'filtered_feature_bc_matrix'))

  seu <- CreateSeuratObject(counts = cur_data, project = sample_name, min.cells = 3, min.features = 200)
  
  seu$orig.ident <- sample_name
  
  return(seu)
})

# Fusionner tous les objets en un seul
seurat.norm_3.pdac_15 <- Reduce(function(x, y) merge(x, y), seurat_list)

# Ajout d'une matadata pour le type d'échantillon (normal ou PDAC)
seurat.norm_3.pdac_15$condition <- ifelse(
  grepl("PDAC", seurat.norm_3.pdac_15$orig.ident, ignore.case = TRUE),
  "PDAC",
  "Norm"
)

# Vérifier combien de cellules par échantillon
table(seurat.norm_3.pdac_15$orig.ident)
table(seurat.norm_3.pdac_15$condition)

library(clusterProfiler)
library(org.Hs.eg.db)
library(dplyr)

# 1) Sélectionner les gènes up dans PDAC (log2FC > 0, p.adj < 0.05)
up_genes_pdac <- de_pdac_vs_norm %>%
  filter(avg_log2FC > 0 & p_val_adj < 0.05) %>%
  pull(gene) %>%
  unique()

# 2) Convertir SYMBOL → ENTREZID pour clusterProfiler
gene_map <- bitr(up_genes_pdac, fromType = "SYMBOL", toType = "ENTREZID", OrgDb = org.Hs.eg.db)
entrez_up <- unique(gene_map$ENTREZID)

# 3) Enrichissement GO (Biological Process)
ego <- enrichGO(
  gene          = entrez_up,
  OrgDb         = org.Hs.eg.db,
  keyType       = "ENTREZID",
  ont           = "BP",           # Biological Process
  pAdjustMethod = "BH",
  qvalueCutoff  = 0.05,
  readable      = TRUE            # convertit ENTIREZID -> SYMBOL
)

# 4) Résultats et visualisation
head(as.data.frame(ego), 10)      # top 10 GO terms

# Barplot
barplot(ego, showCategory = 20, title = "GO BP – Up in PDAC (Macrophages)")

# Dotplot (alternative)
dotplot(ego, showCategory = 20, title = "GO BP – Up in PDAC (Macrophages)")

2.2 Cycle cellulaire

Corriger (ou “compenser”) l’effet du cycle cellulaire peut être une étape clé dans l’analyse single-cell RNA-seq, car le cycle peut introduire une variation biologique non pertinente qui masque les vraies différences de types cellulaires.

D’abord on peut charger les gènes du cycle. Elles sont déjà intégrées dans Seurat:

# Listes intégrées de gènes du cycle cellulaire
s.genes <- cc.genes$s.genes
g2m.genes <- cc.genes$g2m.genes

Note: Si on est dans le cas d’une analyse de données murines, on peux convertir ces listes de gènes avec toupper() ou MouseToHumanOrthologs().

On calcule un score de cycle pour chaque cellule et on visualise cette information avec Dimplot.

Cela ajoute trois colonnes à seurat.norm_3.pdac_15@meta.data : - S.Score : score de phase S - G2M.Score : score de phase G2/M - Phase : phase dominante de la cellule (G1, S ou G2M)

Question: Que peut-on conclure ?

seurat.norm_3.pdac_15 <- CellCycleScoring(
  seurat.norm_3.pdac_15,
  s.features = s.genes,
  g2m.features = g2m.genes,
  set.ident = TRUE
)

table(seurat.norm_3.pdac_15$Phase)
